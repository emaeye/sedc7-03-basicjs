Homework 20: Age calculator


Write a function named calculateAge that: takes 2 arguments: birth year, current year, calculates the age based on those years, outputs the result to the screen like so: "You are NN years old". Call the function three times with different sets of values.



    function calculateAge(birthYear, currentYear) {
        let age = currentYear - birthYear;
        console.log(`You are ${age} years old.`);
    }
    
    let currentYear = new Date().getFullYear();
    
    calculateAge(1980, currentYear);
    calculateAge(1990, currentYear);
    calculateAge(2000, currentYear);
