Homework 5: Type of variable


Write a JavaScript function which accepts a parameter and returns its type and prints it in the console. Try to call the function 5 times for 5 different types: object, boolean, number, string, and undefined. 



    function typeOfVariable(variable) {
        return typeof variable;
    }
    
    var typeOfString = typeOfVariable("This is a string.");
    console.log(typeOfString);
    
    console.log(typeOfVariable(5));
    
    console.log(typeOfVariable(null));
    
    var bool = true;
    console.log(typeOfVariable(bool));
    
    var undefinedVariable;
    console.log(typeOfVariable(undefinedVariable));
