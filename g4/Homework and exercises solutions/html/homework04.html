Homework 4: Convert degrees


Write TWO JavaScript functions to convert temperatures to and from Celsius and Fahrenheit. The first shold convert from °F to °C, the second from °C to °F. Use these formulae: c = (f - 32) * (9 / 5) and f = c * 5 / 9 + 32 (where c = temperature in Celsius and f = temperature in Fahrenheit).
The functions should accept one parameter: the temperature in Fahrenheit for the first function, and the temperature in Celsius for the second.
The functions should print the converted temperature in the console or in an alert window.
Call the functions two times each with these values: 140 °F, 98 °F, 37 °C and 100 °C.



    function convertToCelsius(fahrenheit) {
        var celsius = (fahrenheit - 32) * (5 / 9);
        console.log(`${fahrenheit} °F is the equivalent of ${celsius} °C.`);
    }
    
    function convertToFahrenheit(celsius) {
        var fahrenheit = celsius * 9 / 5 + 32;
        console.log(`${celsius} °C is the equivalent of ${fahrenheit} °F.`);
    }
    
    convertToCelsius(140);
    convertToCelsius(98);
    convertToFahrenheit(37);
    convertToFahrenheit(100);
